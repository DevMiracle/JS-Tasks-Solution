Write a function that receives a number from the user and prints to the console all
numbers between 1 and 1000 that are divisible by the user's number without remainder. 

          
    function printDivisibleNumbers(num) {         
      for (let i = 1; i <= 1000; i++) {       
        if (i % num === 0) {        
          console.log(i);     
        }
      }
    }
